```swift
extension AccountMigration {

    /// Filename for the v3.1 account-layout migration flag.
    public static let v31AccountLayoutFlagFileName = ".v3.1-account-layout-migrated"

    /// One-shot auto-migration of all claude pool accounts to v3.1 layout.
    /// Guarded by a flag file so subsequent invocations are no-ops.
    ///
    /// Best-effort: never throws. Per-account migration errors are written
    /// to stderr but do not block other accounts from being migrated.
    public static func runV31AccountLayoutIfNeeded(homeURL: URL) {
        let fm = FileManager.default
        let flagURL = homeURL.appendingPathComponent(v31AccountLayoutFlagFileName)
        if fm.fileExists(atPath: flagURL.path) { return }
        guard fm.fileExists(atPath: homeURL.path) else { return }

        let acctStore = AccountStore(homeURL: homeURL)
        let envStore = EnvironmentStore(homeURL: homeURL)

        let accounts: [Account]
        do {
            accounts = try acctStore.list(tool: .claude)
        } catch {
            FileHandle.standardError.write(Data(
                "[orrery v3.1 auto-migration] warning: could not list claude accounts: \(error)\n".utf8
            ))
            return
        }

        for acct in accounts {
            do {
                try ClaudeAccountMigration.migrateAccount(
                    acct, accountStore: acctStore, environmentStore: envStore)
            } catch {
                FileHandle.standardError.write(Data(
                    "[orrery v3.1 auto-migration] warning: could not migrate '\(acct.displayName)': \(error)\n".utf8
                ))
            }
        }

        do {
            try Data("v1\n".utf8).write(to: flagURL)
        } catch {
            FileHandle.standardError.write(Data(
                "[orrery v3.1 auto-migration] warning: could not write flag file: \(error)\n".utf8
            ))
        }
    }
}
```

## Task 8: Simplify shell function `use)` — drop v3.0.4 fallback

**Files:**

Plan 3's `use)` case had a fallback path for non-v3.1 claude accounts. After Plan 4 T1 (auto-migrate), no claude account is ever in v3.0.4 layout. The fallback is dead code that could mask bugs.

New behavior:
- For claude: `_account-dir` always succeeds (auto-migrated). Always export CLAUDE_CONFIG_DIR.
- For codex/gemini: pass through to `command orrery-bin use` (no per-account dir concept for those tools).
- If something WAS wrong (e.g., `_account-dir` fails for a claude account): surface the error visibly. Don't silently fall back.

- [ ] **Step 1: Update the `use)` case**

Replace the v3.0.4 fallback with explicit error surfacing. For non-claude tools, route directly to `command orrery-bin use`.

```bash
            use)
              # Help / version requests bypass.
              for _a in "${@:2}"; do
                case "$_a" in
                  -h|--help|--version) command orrery-bin "$@"; return $?; ;;
                esac
              done
              shift
              # Detect tool: --codex / --gemini route directly to orrery-bin
              # (those have no per-account dir concept). Anything else (default
              # or --claude) goes through the v3.1 fast-path.
              for _a in "$@"; do
                case "$_a" in
                  --codex|--gemini)
                    command orrery-bin use "$@"
                    return $?
                    ;;
                esac
              done
              # claude (explicit or default).
              local _dir

                export CLAUDE_CONFIG_DIR="$_dir"
                echo "orrery: CLAUDE_CONFIG_DIR=$_dir" >&2
              else
                # _account-dir failed for a claude account — should not happen
                # post-auto-migrate. Surface the error visibly.
                echo "orrery: use failed:" >&2
                echo "$_dir" >&2
                return 1
              fi
              ;;
```
